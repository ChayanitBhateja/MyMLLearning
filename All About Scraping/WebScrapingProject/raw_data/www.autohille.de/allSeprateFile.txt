function jumpToElement (elementId, offsetAdjustment = 0) {
    let $ element = document.getElementById (elementId);
    if ($ element! == null) {
        let top = $ element.offsetTop;
        top + = offsetAdjustment;
        window.scrollTo ({top: top, left: 0, behavior: "smooth"});
